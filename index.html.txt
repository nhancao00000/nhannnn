const homeUser = JSON.parse(localStorage.getItem("fooddash_user") || "null");
if (homeUser) {
    document.getElementById("loginNavLink").classList.add("hidden");
    document.getElementById("userGreeting").textContent = `Xin chào, ${homeUser.full_name}`;
    document.getElementById("userGreeting").classList.remove("hidden");
    document.getElementById("logoutNavLink").classList.remove("hidden");
}
if (homeUser && homeUser.role === "admin") {
    document.getElementById("adminNavLink").style.display = "inline-flex";
}
document.getElementById("logoutNavLink").addEventListener("click", (event) => {
    event.preventDefault();
    localStorage.removeItem("fooddash_user");
    window.location.reload();
});
